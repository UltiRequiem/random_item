# random_item

> Get a random item from an array

## Usage

```typescript
import {
  randomMultipleItems,
  randomItem,
} from "https://deno.land/x/random_item/mod.ts";

randomItem(["🐴", "🦄", "🌈"]);
//=> '🦄'

randomMultipleItems(["🐴", "🦄", "🌈"], 2);
//=> ['🌈', '🦄']
```
